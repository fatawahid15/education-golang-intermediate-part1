package main

import (
	"fmt"
	"time"
)

func main() {

	// 00:00:00 UTC on Jan 1 1970

	TimeNow := time.Now()
	unixTime := TimeNow.Unix()
	fmt.Println("Current Unix Time:", unixTime)

	t := time.Unix(unixTime, 0)
	fmt.Println(t)
	fmt.Println("Time:", t.Format("06-01-02"))
}
