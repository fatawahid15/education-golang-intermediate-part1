package main

import (
	"errors"
	"fmt"
)

func main() {

	err := doSomething()
	if err != nil {
		fmt.Println(err)
		return // everytime we check for an error we have to return so the rest of the code under this if else statement/return doesn't get executed
	}
	fmt.Println("Operation completed successfully!")

}

type customError struct {
	code    int
	message string
	er      error
}

// Error returns thej error message. Implementing Error() method of error interface
func (e *customError) Error() string {
	return fmt.Sprintf("Error %d: %s, %v\n", e.code, e.message, e.er)
}

			code:    500,
			message: "Something went wrong",

		}

}

func doSomethingElse() error {
	return errors.New("internal error")
